local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- ตั้งค่า Endpoint ของ API
local endpoint = "https://roblox-l5m8.onrender.com/update"

-- ฟังก์ชันค้นหา Text อย่างปลอดภัย โดยรอโหลดสูงสุด 90 วินาที
local function safeFindText(base, path, maxWaitTime)
	maxWaitTime = maxWaitTime or 90
	local elapsed = 0
	local obj = base

	for _, part in ipairs(path) do
		local found = nil
		repeat
			found = obj:FindFirstChild(part)
			if not found then
				wait(1)
				elapsed += 1
			end
		until found or elapsed >= maxWaitTime

		if not found then
			return "Unknown"
		end
		obj = found
	end

	if obj:IsA("TextLabel") or obj:IsA("TextBox") or obj:IsA("TextButton") then
		return obj.Text
	end

	return "Unknown"
end

-- ฟังก์ชันดึงข้อมูลจาก GUI
local function getData()
	local gui = player:WaitForChild("PlayerGui")

	-- ดึงจำนวนเงิน
	local inventory = gui:WaitForChild("Inventory"):WaitForChild("CanvasGroup")
	local amountLabel = inventory.Main.Body.Cash.Main.Amount
	local cash = tonumber(amountLabel.Text) or 0

	-- ดึงชื่อเซิร์ฟเวอร์
	local serverName = safeFindText(gui, {
		"TopbarStandard", "Holders", "Left", "Widget",
		"IconButton", "Menu", "IconSpot", "Contents",
		"IconLabelContainer", "IconLabel"
	}, 90)

	-- ดึงจำนวนผู้เล่น และชื่อ
	local playerCount = #Players:GetPlayers()
	local username = player.Name

	return {
		username = username,
		cash = cash,
		playercount = playerCount,
		servername = serverName
	}
end

-- ฟังก์ชันส่งข้อมูลไป API พร้อมระบบ retry
local function sendData()
	local data = getData()
	local jsonData = HttpService:JSONEncode(data)

	print("กำลังส่งข้อมูลไปยัง API:", jsonData)

	local success, response = pcall(function()
		request({
			Url = endpoint,
			Method = "POST",
			Headers = {
				["Content-Type"] = "application/json"
			},
			Body = jsonData
		})
	end)

	if success then
		print("ส่งข้อมูลสำเร็จ")
	else
		warn("ส่งข้อมูลล้มเหลว:", response)
		wait(2)
		sendData()
	end
end

-- เริ่มส่งข้อมูลอัตโนมัติ
task.spawn(function()
	wait(1) -- รอ GUI อื่นๆ โหลด
	sendData()

	while true do
		wait(1)
		sendData()
	end
end)
